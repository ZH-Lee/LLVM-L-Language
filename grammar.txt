Grammar:

Type
        :   int
        |   double

Constant
        :   [0-9]*.[0-9]*

Identifier
        :   [a-zA-Z_][a-zA-Z0-9]*

binary_operator
        :   '+'
        |   '-'
        |   '*'
        |   '/'
        |   '='

primary_expression
        :   (Identifier|Constant) (binary_operator (Identifier|Constant) )*
        |   function_call '(' (Identifier)* ')'
        |   unary Identifier
        |   return Identifier ';'

variable_define_expression
        :   Type Identifier '=' primary_expression

function_call
        :   Type Identifier '=' function_call '(' (Identifier)* ')' ';'
        |   function_call '(' Identifier (,Identifier)* ')' ';'


function_define_expression
        :   def Identifier '(' Type Identifier (, Type Identifier)* ')' \
                   '{' primary_expression '}' ';'
